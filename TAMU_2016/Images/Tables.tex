% wOBA Equation
\begin{align*}
	wOBA = \dfrac{.69 * uBB + .72 * HBP + .89 * 1B + 1.27 * 2B + 1.62 * 3B + 2.10 * HR}{AB + BB - IBB + SF + HBP}
\end{align*}

% BABIP Equation
\begin{align*}
	BABIP = \dfrac{H - HR}{AB - K - HR + SF}
\end{align*}

% wRAA Equation
\begin{align*}
	wRAA = [\dfrac{wOBA - league wOBA}{wOBA scale}] * PA
\end{align*}
